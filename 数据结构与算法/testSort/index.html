function sort(arr) { const len = arr.length; if (len< 2) return arr; let gap = 1; while (gap < len / 5) { gap = gap * 5 + 1; } for (gap; gap > 0; gap = Math.floor(gap / 5)) { for (let i = gap; i < len; i++) { let j = i - gap, temp = arr[i]; while (j >= 0 && arr[j] > temp) { arr[j + gap] = arr[j]; j = j - gap; } arr[j + gap] = temp; } } return arr; } const res = sort([3, 44, 38, 5, 47, 15, 36, 26, 27, 2, 46, 4, 19, 50, 48]) console.info('res:', res)
